#0.Use npx to initialize and set up your app
npx create-react-app my-react-app
cd my-react-app

#1.Create a Single-File React App
import React from 'react';
function App() {
  return (
    <div>
      <h1>Hello, ByteXL!</h1>
      <p>Practice 1 - Cloud Deployment: Single File React App.</p>
    </div>
  );
}
export default App;

#2.Write a Multi-Stage Dockerfile
# Stage 1: Build React app with Node.js
FROM node:18 AS build
WORKDIR /app
COPY package.json package-lock.json ./
RUN npm install
COPY . .
RUN npm run build

# Stage 2: Serve with Nginx
FROM nginx:alpine
COPY --from=build /app/build /usr/share/nginx/html
EXPOSE 80
CMD ["nginx", "-g", "daemon off;"]

#3.Add a .dockerignore File
node_modules
build
.dockerignore
Dockerfile
npm-debug.log

#4.Build and Run the Docker Image Locally
docker build -t my-react-app .
docker run -p 80:80 my-react-app
